cmake_minimum_required ( VERSION 3.16 )
project ( tester )
set ( CMAKE_CXX_STANDARD 17 )
set ( CMAKE_CXX_STANDARD_REQUIRED ON )
get_filename_component ( projectsFolder ${PROJECT_SOURCE_DIR} DIRECTORY )
set ( BOOST_ROOT ${projectsFolder}/boost_1_74_0 )
set ( Boost_USE_STATIC_LIBS ON )
find_package ( Boost COMPONENTS REQUIRED system thread date_time )
include_directories ( ${Boost_INCLUDE_DIRS} )
add_definitions ( -DUNICODE )
if ( WIN32 )
    add_definitions ( -D_WIN32_WINNT=0x0601 )
endif ()
file ( GLOB sources *.h *.cpp *.def 1c/*.h 1c/*.cpp )
add_library ( ${PROJECT_NAME} SHARED ${sources} )
target_link_libraries ( ${PROJECT_NAME} ${Boost_LIBRARIES} )
